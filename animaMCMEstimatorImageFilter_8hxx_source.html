868 ::SpecificModelEstimation(MCMPointer &mcmValue, std::vector <double> &observedSignals, itk::ThreadIdType threadId,
869  double &aiccValue, double &b0Value, double &sigmaSqValue)
870 {
871  // Finally, we're done with ball and zeppelin, an example of what's next up with multi-tensor
872  // - First create model
873  MCMCreatorType *mcmCreator = m_MCMCreators[threadId];
874  // Other params are supposed to be already initialized from trunk estimation
875  mcmCreator->SetCompartmentType(Tensor);
876 
877  MCMPointer mcmUpdateValue = mcmCreator->GetNewMultiCompartmentModel();
878  MCMPointer mcmOptimalModel = mcmCreator->GetNewMultiCompartmentModel();
879 
880  CostFunctionBasePointer cost = this->CreateCostFunction(observedSignals,mcmUpdateValue);
881 
882  // - Update multi-tensor model against observed signals
883  unsigned int dimension = mcmUpdateValue->GetNumberOfParameters();
884  ParametersType p(dimension);
885  MCMType::ListType workVec(dimension);
886  itk::Array<double> lowerBounds(dimension), upperBounds(dimension);
887 
888  workVec = mcmUpdateValue->GetParameterLowerBounds();
889  for (unsigned int i = 0;i < dimension;++i)
890  lowerBounds[i] = workVec[i];
891 
892  workVec = mcmUpdateValue->GetParameterUpperBounds();
893  for (unsigned int i = 0;i < dimension;++i)
894  upperBounds[i] = workVec[i];
895 
896  double optimalCostValue = 0;
897  double optimalB0Value = 0;
898  double optimalSigmaSqValue = 0;
899 
900  unsigned int numIsoCompartments = mcmUpdateValue->GetNumberOfIsotropicCompartments();
901  unsigned int numCompartments = mcmUpdateValue->GetNumberOfCompartments();
902 
903  unsigned int numNonIsoCompartments = numCompartments - numIsoCompartments;
904  SequenceGeneratorType ldsAngles(3 * numNonIsoCompartments);
905  MCMType::ListType lowerBoundsAngleSampling(3 * numNonIsoCompartments,0.0);
906  MCMType::ListType upperBoundsAngleSampling(3 * numNonIsoCompartments,2.0 * M_PI);
907 
908  for (unsigned int i = 0;i < numNonIsoCompartments;++i)
909  {
910  upperBoundsAngleSampling[numNonIsoCompartments + i] = 1.0;
911  upperBoundsAngleSampling[2 * numNonIsoCompartments + i] = 1.0;
912  }
913 
914  ldsAngles.SetLowerBounds(lowerBoundsAngleSampling);
915  ldsAngles.SetUpperBounds(upperBoundsAngleSampling);
916  std::vector <double> sampledData;
917 
918  for (unsigned int restartNum = 0;restartNum < 3 * m_NumberOfRandomRestarts;++restartNum)
919  {
920  // - Now the tricky part: initialize from previous model, handled somewhere else
921  this->InitializeModelFromSimplifiedOne(mcmValue,mcmUpdateValue);
922 
923  sampledData = ldsAngles.GetNextSequenceValue();
924 
925  for (unsigned int i = numIsoCompartments;i < numCompartments;++i)
926  {
927  unsigned int index = i - numIsoCompartments;
928  // Random alpha initialization
929  mcmUpdateValue->GetCompartment(i)->SetPerpendicularAngle(sampledData[index]);
930 
931  // Takes care of initializing d2=l2-l3 (which is 0 in Zeppelin) to a positive realistic value
932  double zeppelinAxDiff = mcmUpdateValue->GetCompartment(i)->GetAxialDiffusivity();
933  double zeppelinRadDiff = mcmUpdateValue->GetCompartment(i)->GetRadialDiffusivity1();
934 
935  double w1 = sampledData[numNonIsoCompartments + index];
936  mcmUpdateValue->GetCompartment(i)->SetRadialDiffusivity1(w1 * zeppelinRadDiff + (1 - w1) * zeppelinAxDiff);
937 
938  double w2 = sampledData[2 * numNonIsoCompartments + index];
939  mcmUpdateValue->GetCompartment(i)->SetRadialDiffusivity2(w2 * zeppelinRadDiff + (1 - w2) * 1e-5);
940  }
941 
942  workVec = mcmUpdateValue->GetParametersAsVector();
943  for (unsigned int i = 0;i < dimension;++i)
944  p[i] = workVec[i];
945 
946  double costValue = this->PerformSingleOptimization(p,cost,lowerBounds,upperBounds);
947 
948  if ((costValue < optimalCostValue)||(restartNum == 0))
949  {
950  // - Get estimated data
951  for (unsigned int i = 0;i < dimension;++i)
952  workVec[i] = p[i];
953 
954  mcmOptimalModel->SetParametersFromVector(workVec);
955 
956  this->GetProfiledInformation(cost,mcmOptimalModel,optimalB0Value,optimalSigmaSqValue);
957  optimalCostValue = costValue;
958  }
959  }
960 
961  aiccValue = this->ComputeAICcValue(mcmOptimalModel,optimalCostValue);
962  mcmValue = mcmOptimalModel;
963  b0Value = optimalB0Value;
964  sigmaSqValue = optimalSigmaSqValue;
965 }
